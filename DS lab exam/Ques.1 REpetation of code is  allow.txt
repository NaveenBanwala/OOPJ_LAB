#include <stdio.h>
int main(){
    int a[]={1, 2, 4, 10, 20, 40}; 
    int N=sizeof(a) / sizeof(a[0]); 
    int count=0;
    int i=0;
    while(i!=N){
        for (int j = 0; j < N; j++) {
            if (a[i] * a[j] == 40) {
                printf("(%d,%d)\n",a[i],a[j]);

        }
        i++;
